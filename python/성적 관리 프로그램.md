def grade(average):
    if average >= 90:
        grade = 'A'
    elif average >= 80:
        grade = 'B'
    elif average >= 70:
        grade = 'C'
    elif average >= 60:
        grade = 'D'
    else:
        grade = 'F'
    return grade

def remove(member):
    temp = False
    if len(member) != 0:
        try:
            removeId = int(input("StudentID: "))
            for i in range(len(member)):
                if removeId == member[i][0]:
                    del member[i]
                    print("Student removed.")
                    temp = True
                    break
            if temp == False:
                print("NO SUCH PERSON.")
        except:

def quit(member):
    saveData = input("Save data?[yes/no] ").lower()
    temp = False
    if saveData == 'yes':
        member.sort(key=lambda x: x[4], reverse=True)
        filename = input("File name: ")
        with open(filename, "w") as fw:
            for i in member:
                data = "%d\t%s\t%d\t%d\n" % (i[0], i[1], i[2], i[3])
                fw.write(data)
        temp = True
    elif saveData == 'no':
        temp = True
    else:
        print("잘못 입력하셨습니다.")
    return temp

def readdata(data):
    member = []
    for i in data:
        temp = i.split('\t')
        temp[0], temp[2], temp[3] = int(temp[0]), int(temp[2]), int(temp[3].strip())
        average = float((temp[2] + temp[3]) / 2)
        temp.append(average)
        temp.append(grade(average))
        member.append(temp)
    return member

def main():
    import sys
    import os
    if os.path.exists(sys.argv[1]) == True:
        with open(sys.argv[1], "r") as fr:
            data = fr.readlines()
            member = readdata(data)
    else:
        with open("students.txt", "r") as fr:
            data = fr.readlines()
            member = readdata(data)
    printFrame()
    show(member)
    printendFrame()

    while True:
        cmd = input("# ").replace(" ", "").lower()

            printFrame()

            printendFrame()
        elif cmd == "search":
            search(member)
        elif cmd == "searchname":
            searchname(member)
        elif cmd == "changescore":
            changescore(member)
        elif cmd == "changename":
            changename(member)
        elif cmd == "add":
            add(member)
        elif cmd == "searchgrade":
            searchgrade(member)
        elif cmd == "remove":
            remove(member)
        elif cmd == "quit":
            temp = quit(member)
            if temp == True:
                break
        else:
            continue

if __name__=='__main__':
